# define a tweets dictionary
tweet_dict = {}

#define lists to append line data into
tweets, tweet_id, tweet_text,favorite_count,retweet_count,created_at,followers_count,user_likes,hashtags,user_mentions  = [],[],[],[],[],[],[],[],[],[]

#read file line, loop over the lines
with open('tweet_json.txt') as file:
    
    for tweet in file:
        tweet = json.loads(tweet)
        tweet_id.append(tweet['id'])
        tweet_text.append(tweet['full_text'])
        favorite_count.append(tweet['favorite_count'])
        retweet_count.append(tweet['retweet_count'])
        created_at.append(tweet['created_at'])
        followers_count.append(tweet['user']['followers_count'])
        user_likes.append(tweet['user']['favourites_count'])
        
        if len(tweet['entities']['hashtags']) != 0:
            hashtags.append(tweet['entities']['hashtags'][0]['text'])
        else:
            hashtags.append('-')
        if len(tweet['entities']['user_mentions']) != 0:
            user_mentions.append(tweet['entities']['user_mentions'][0]['screen_name'])
        else:
            user_mentions.append('-')


tweet_dict = {
            'tweet_id':tweet_id,
            'tweet_text':tweet_text,
            'favorite_count':favorite_count,
            'retweet_count':retweet_count,
            'created_at':created_at,
            'followers_count':followers_count,
            'user_likes':user_likes,
            'hashtags':hashtags,
            'user_mentions':user_mentions
            }

tweet_df = pd.DataFrame(data=tweet_dict)

#save the dictionary into a csv file
tweet_df.to_csv('twitter_archive_master.csv', index=False)

def clean_tweet_text(text):
    #remove user mentions and hashtags
    clean_text = re.sub(r"@(\w+)",'', str(text))
    clean_text = re.sub(r"#(\w+)",'',clean_text)
    
    #remove dog names - Tweets introduce names using 'This is', 'Meet', 'Say hello to'
    clean_text = re.sub(r"This is (\w+).|Meet (\w+).| Say hello to (\w+).",'',clean_text)
    
    #remove any url in the tweet
    clean_text = re.sub(r'(?i)\b((?:https?://|www\d{0,3}[.]|[a-z0-9.\-]+[.][a-z]{2,4}/)(?:[^\s()<>]+|\(([^\s()<>]+|(\([^\s()<>]+\)))*\))+(?:\(([^\s()<>]+|(\([^\s()<>]+\)))*\)|[^\s`!()\[\]{};:\'".,<>?«»“”‘’]))', '', clean_text)
    
    #removing any other punctuation or any other non-alphabetical characters
    clean_text = re.sub(r"[.,;:$&*|?'-]",' ',clean_text)
    clean_text = re.sub(r"(\d+/\d+)",'',clean_text)
    
    return clean_text

def calculate_rating(rating):
    
    if rating:
        rated = int(rating.split('/')[0])
        rated_out_of = int(rating.split('/')[1])
        
        #catch division by 0 error
        try:
            calculated_rating = int((rated/rated_out_of)*100)
        except ZeroDivisionError:
            calculated_rating = 0
            
        return calculated_rating
    else:
        return 0

def analyze_sentiment(tweet):
    analysis = TextBlob(tweet)
    if analysis.sentiment.polarity > 0:
        return 'Positive'
    elif analysis.sentiment.polarity == 0:
        return 'Neutral'
    else:
        return 'Negative'
